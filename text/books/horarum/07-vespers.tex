\sectionTitle{Ad Vespertinum}

% Psalm 140
\prayerText{\firstLetter{D}òmine, clamávi ad te, exáudi me~; \symbolBreak exáudi me, Dòmine. \symbolBreak Dòmine, clamávi ad te, exáudi me~; \symbolBreak intènde voċi meae, \symbolBreak quum clamávero ad te. \symbolBreak Dirigátur oráçio mea \symbolBreak sicut inċènsum in conspèctu tuo~; \symbolBreak eleváçio mànuum meárum \symbolBreak sacrifìċium vespertínum. \symbolBreak Exáudi me, Dòmine.}

\prayerVersicle{Pone, Dòmine, custódiam ori meo, et óstium çircumstànçiae lábiïs meïs.}

\prayerResponse{Non declínes cor meum in vèrba malìçiae, ad excusàndas excusaçiónes in peccátïs.}

\prayerVersicle{cum homìnibus operàntibus iniquitátem, et non communicábo cum eléctïs eórum.}

\prayerResponse{Corrìpiet me justus in mizericórdia, et increpábit me~; óleum autem peccatóris non impìngüet càput meum.}

\prayerVersicle{Quóniam adhüc et oráçio mea in benepláċitïs eórum; absórpti sunt jüncti pètrae júdices eórum.}

\prayerResponse{Áudient vèrba mea quóniam potuérunt; sicut crassitúdo tèrrae erùpta est super tèrram, dissipáta sunt òssua nostra sècus infèrnum.}

\prayerVersicle{Quia ad te, Dòmine, Dòmine òculi mei~; in te sperávi, non áuferas ànimam meam.}

\prayerResponse{Custódi me a làqueo, quem statuvérunt mìhi, et a scàndalïs operàntium iniquitátem.}

\prayerVersicle{Cadent in retìculo ejus peccatóres~; singuláriter sum ego donec trànseam.}

% Psalm 141
\prayerResponse{Voċe mea ad Dòminum clamávi, voċe mea ad Dòminum deprecátus sum.}

\prayerVersicle{Effùndo in conspèctu ejus deprecaçiónem meam, et tribulaçiónem meam ànte ìpsum pronúnçio}

\prayerResponse{In defiċièndo e me spíritum meum, et tu cognovìsti sémitas meas.}

\prayerVersicle{In via hac, qua ambulábam, absconduérunt làqueum mìhi.}

\prayerResponse{Considerábam ad dèxteram, et vidébam, et non èrat qui cognósceret me.}

\prayerVersicle{Perívit fuġéla a me, et non est qui requírat ànimam meam.}

\prayerResponse{Clamávi ad te, Dòmine~; dixi, «~Tu es spes mea, pórçio mea in tèrra vivèntium.~»}

\prayerVersicle{Intènde ad deprecaçiónem meam, quia humiliátus sum nìmis.}

\prayerResponse{Líbera me a persequèntibus me, quia confortáti sunt super me.}

% 10
\prayerVersicle{Edúce de custódia ànimam meam ad confitèndum nómini tuo.}

\prayerResponse{me exspèctant justi donec retrìbuas mìhi.}

% Psalm 129
\prayerVersicle{De profùndïs clamávi ad te, Dòmine. Dòmine, exáudi voċem meam.}

\prayerResponse{Fiant aures tuae intendèntes in voċem deprecaçiónis meae.}

\prayerVersicle{¿~Si iniquitátes observáverïs, Dòmine, Dòmine, quìs sustinébit~? Quia àpud te propiçiáçio est.}

\prayerResponse{Pròpter leġem tuam sustìnui te, Dòmine~; sustìnuit ànima mea in vèrbo ejus; sperávit ànima mea in Dòmino}

\prayerVersicle{A custódia matutína usque ad nòctem, speret Ìsraël in Dòmino.}

\prayerResponse{Quia àpud Dòminum mizericórdia et copiósa àpud eum redèmpçio, et ìpse rèdimet Israélem ex òmnibus iniquitátibus ejus.}

\prayerResponse{Quóniam confirmáta est super nos mizericórdia ejus, et vèritas Dòmini mànet in aetèrnum.}
