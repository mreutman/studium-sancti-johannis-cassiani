\newcommand{\prayerFeriaI}{
  \prayerDayofWeek
  {Feria Secunda}
  {ad angelos}
  {\firstLetter{D}eus, qui ineffábili providènçia sanctos ànġelos tuos ad nostram custódiam mìttere dignáris: larġíre supplìċibus tuïs, et eórum semper protecçióne defèndi, et aetèrna soċietáte gaudére. Per Dòminum nostrum Jesum Christum, Fílium tuum, qui tecum vivit et regnat in unitáte Spíritus Sancti, Deus, per òmnia sáecula saeculórum.} 
  {Sancti ànġeli, custódes nostri, defèndite nos in próelio, ut non pereámus in tremèndo judìċio.}
}

\newcommand{\prayerFeriaII}{
  \prayerDayofWeek
  {Feria Terçia}
  {ad Johannem Baptistam}
  {\firstLetter{P}raesta, quáesumus, omnípotens Deus: ut família tua per viam salútis inċédat; et beáti Johànnis Praecursóris hortamènta sectàndo, ad eum quem praedíxit, secúra pervèniat, Dòminum nostrum Jesum Christum, Fílium tuum.} 
  {Ora pro nobïs, sancte Johànnes Baptìsta, ut dìgni effiċiámur promissiónibus Christi.}
}

\newcommand{\prayerFeriaIII}{
  \prayerDayofWeek
  {Feria Quarta}
  {ad Mariam Virġinem}
  {\firstLetter{D}eus, qui salútis aetèrnae, beátae Maríae virġinitáte fecùnda, humáno ġèneri práemia praestitìsti: tríbue, quáesumus, ut ìpsam pro nobïs interċédere sentiámus, per quam merúimus Fílium tuum auctórem vitae susċìpere, qui tecum vivit et regnat in unitáte Spíritus Sancti, Deus, per òmnia sáecula saeculórum.} 
  {Ora pro nobïs, sancta Dei Ġènetrix, ut dìgni effiċiámur promissiónibus Christi.}
}
