\newcommand{\psalmChapter}[2]{%
  \needspace{1\baselineskip}
  \begin{nscenter}

    \begin{spacing}{1.25}
    \psalmTitle{#1}
    \end{spacing}
  }
  \end{nscenter}

  \medskip

  \psalmBody{#2}
}

  \end{nscenter}

  \medskip

  \psalmBody{#3}
