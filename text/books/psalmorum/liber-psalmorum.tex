\newcommand{\verseFirstLetter}[1]
  {{\sizeVerseFirstLetter #1}}

\newcommand{\verseNum}[1]
  {\noindent{{\textup{\textsuperscript{\sizeVerseNumber #1.}}}}}

\newcommand{\psalmTitle}[1]{%
  \begin{nscenter}
  \bfseries\sizePsalmTitle PSALMUS #1.
  \end{nscenter}
}

\newcommand{\psalmBody}[1]{%
  \setlength{\parindent}{0cm}
  \textsc{\sizePsalmBody #1}
}

% PUBLIC - - - - - - - - - - - - -
\newcommand{\psalmChapter}[2]{%
  \needspace{7\baselineskip}
  \psalmTitle{#1}

  \smallskip

  \psalmBody{#2}
}

% PUBLIC - - - - - - - - - - - - - - - - - - - -
\newcommand{\psalmChapterWithInscription}[3]{%

  %\smallskip

  %\begin{nscenter}
  %\textsc{\sizePsalmInscription #2} % Chapter inscription

    \begin{minipage}{0.75\textwidth}
    \begin{nscenter}
    \textsc{\sizePsalmInscription #2}
    \end{nscenter}
    \end{minipage}
  }

  \medskip

  \psalmBody{#3}
}
