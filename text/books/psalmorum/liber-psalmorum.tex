\newcommand{\verseFirstLetter}[1]
  {{\sizeVerseFirstLetter #1}}

\newcommand{\verseNum}[1]
  {\noindent{{\textup{\textsuperscript{\itshape\sizeVerseNumber #1.}}}}}

\newcommand{\verseNumInscription}[1]
  {\noindent{{\textup{\textsuperscript{\itshape\sizeVerseNumberInscription #1.}}}}}

\newcommand{\psalmTitle}[1]{%
  \begin{nscenter}
  {\bfseries\sizePsalmTitle PSALMUS \RomanNumeralCaps{#1}.}
  \end{nscenter}
}

\newcommand{\psalmBody}[1]{%
  \setlength{\parindent}{0cm}
  \textsc{\sizePsalmBody #1}

  \skipAfterPsalm
}

% PUBLIC - - - - - - - - - - - - -
\newcommand{\psalmChapter}[2]{%
  \needspace{1\baselineskip}
  \psalmTitle{#1}

  %\smallskip

  \psalmBody{#2}
}

% PUBLIC - - - - - - - - - - - - - - - - - - - -
\newcommand{\psalmChapterWithInscription}[3]{%
  \needspace{2\baselineskip}
  %\psalmTitle{#1}

  %\smallskip

  \begin{nscenter}
  \parbox{0.8\textwidth}{%
    %\begin{nscenter}
    \begin{spacing}{1.25}
    {\bfseries\sizePsalmTitle \textsc{Psalmus} {\large\addfontfeature{LetterSpace=10.0} \RomanNumeralCaps{#1}.}}~~~\textsc{\itshape\sizePsalmInscription #2}
    \end{spacing}
    %\end{nscenter}
  }
  \end{nscenter}

  \medskip

  \psalmBody{#3}
}

% PUBLIC - - - - - - - - -
\newcommand{\prayer}[2]{%
  \psalmChapter{#1}{#2}
}
