\psalmChapterInscription{121}
{\verseNumInscription{1}~Cànticum gràduum}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\firstLetter{L}aetátus sum in hīs quae dicta sunt mìhi, «~In domum Dòmini íbimus.~»
\verseNum{2}~Stantes èrant pèdes nostri in átriīs tuīs, Hierozólyma.
\verseNum{3}~Hierozólyma, quae aedificátur ut çívitas, cujus partiċipáçio ejus in idìpsum.
\verseNum{4}~Illūc ènim asċensérunt trìbūs, trìbūs Dòmini, testimónium Israélis, ad confitèndum nómini Dòmini.
\verseNum{5}~Quia illic sēdérunt sedes in judìċio, sedes super domum Davídis.
\verseNum{6}~Rogáte quae ad paċem sunt Hierozólymae, et abundànçia diligèntibus te.
\verseNum{7}~Fiat pax in virtúte tua, et abundànçia in tùrribus tuīs.
\verseNum{8}~Pròpter fratres meos et pròximos meos, loquébar paċem de te.
\verseNum{9}~Pròpter domum Dòmini Dei nostri, quaesívi bona tìbi.}
