\psalmChapterWithInscription{144}
{\verseNumInscription{1} Laudáçio ìpsi Davídi}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{E}xaltábo te, Deus meus rex, et benedícem nómini tuo in sáeculum, et in sáeculum sáeculi. 
\verseNum{2}~Per sìngulos dies benedícem tìbi, et laudábo nomen tuum in sáeculum, et in sáeculum sáeculi. 
\verseNum{3}~Magnus Dòminus, et laudábilis nìmis, et magnitúdinis ejus non est finis. 
\verseNum{4}~Ġeneráçio et ġeneráçio laudábit ópera tua, et potènçiam tuam pronunçiábunt. 
\verseNum{5}~Magnifiċènçiam glóriae sanctitátis tuae loquèntur, et mirabìlia tua narrábunt. 
\verseNum{6}~Et virtútem terribìlium tuórum dicent, et magnitúdinem tuam narrábunt. 
\verseNum{7}~Memóriam abundànçiae süavitátis tuae eructábunt, et justìçia tua exsultábunt. 
\verseNum{8}~Mizerátor et mizèricors Dòminus~; pàtiens, et mùltum mizèricors. 
\verseNum{9}~Süávis Dòminus univèrsïs, et mizeraçiónes ejus super òmnia ópera ejus. 
\verseNum{10}~Confiteàntur tìbi, Dòmine, òmnia ópera tua, et sancti tui benedícant tìbi. 
\verseNum{11}~Glóriam regni tui dicent, et potènçiam tuam loquèntur, 
\verseNum{12}~ut notam fàċiant fíliïs hòminum potènçiam tuam, et glóriam magnifiċènçiae regni tui. 
\verseNum{13}~Regnum tuum regnum òmnium saeculórum, et domináçio tua in òmni ġeneraçióne et ġeneraçiónem. 
\verseNum{14}~Fidélis Dòminus in òmnibus vèrbïs suïs, et sanctus in òmnibus opèribus suïs. 
\verseNum{15}~Àllevat Dòminus òmnes qui còrruunt, et érigit òmnes elísos. 
\verseNum{16}~Òculi òmnium in te sperant, Dòmine, et tu das escam illórum in tèmpore opportúno. 
\verseNum{17}~Àperïs tu mànum tuam, et imples omne ànimal benedicçióne. 
\verseNum{18}~Justus Dòminus in òmnibus viïs suïs, et sanctus in òmnibus opèribus suïs. 
\verseNum{19}~Prope est Dòminus òmnibus invocàntibus eum, òmnibus invocàntibus eum in veritáte. 
\verseNum{20}~Voluntátem timèntium se fàċiet, et deprecaçiónem eórum exáudiet, et sàlvos fàċiet eos. 
\verseNum{21}~Custódit Dòminus òmnes diligèntes se, et òmnes peccatóres dispèrdet. 
\verseNum{22}~Laudaçiónem Dòmini loquétur os meum, et benedícat òmnis càro nómini sancto ejus in sáeculum, et in sáeculum sáeculi. 
}
