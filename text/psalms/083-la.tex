\psalmChapterInscription{83}
{\verseNumInscription{1}~In finem pro torculáribus fíliīs Córachi Psàlmus}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\firstLetter{Q}uam dilécta tabernácula tua Dòmine virtútum.
\verseNum{3}~Concupíscit et défiċit ànima mea in átria Dòmini~; cor meum et carnis mea exsultavérunt in Deum vivum.
\verseNum{4}~Ètenim passer invénit sìbi domum, et turtur nidum sìbi, ubi ponat pullos suos~; altária tua, Dòmine virtútum, rex meus, et Deus meus.
\verseNum{5}~Beáti qui hábitant in domu tua, Dòmine~; in sáecula saeculórum laudábunt te.
\verseNum{6}~Beátus vìr cujus est auxìlium a te~; asċensiónes in corde suo dispósuit,
\verseNum{7}~in valle lacrymárum, in loco quem pósuit.
\verseNum{8}~Ètenim benedicçiónem dabit leġislátor~; ibunt de virtúte in virtútem~; vidébitur Deus deórum in Zióne.
\verseNum{9}~Dòmine Deus virtútum, exáudi oraçiónem meam~; áuribus pèrċipe, Deus Jacóbi.
\verseNum{10}~Protèctor noster, àspiċe, Deus, et rèspiċe in fàċiem christi tui.
\verseNum{11}~Quia mèlior est dies una in átriīs tuīs super mìllia~; elégi abjèctus esse in domu Dei mei maġis quam habitáre in tabernáculīs peccatórum.
\verseNum{12}~Quia mizericórdiam et veritátem díligit Deus~; gráziam et glóriam dabit Dòminus.
\verseNum{13}~Non privábit bonīs eos qui àmbulant in innoċènçia~; Dòmine virtútum, beátus hòmo qui sperat in te.}
