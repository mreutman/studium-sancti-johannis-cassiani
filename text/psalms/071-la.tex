\psalmChapterInscription{71}
{\verseNumInscription{1} In Salomónem}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\verseFirstLetter{D}eus, judìċium tuum reġi da, et justìçiam tuam fílio reġis~; 
\verseNum{3}~judicáre pòpulum tuum in justìçia, et páuperes tuos in judìċio. 
\verseNum{4}~Susċìpiant mòntes paċem pòpulo, et colles justìçiam. 
\verseNum{5}~Judicábit páuperes pòpuli, et sàlvos fàċiet fílios páuperum, et humiliábit calumniatórem. 
\verseNum{6}~Et permanébit cum sole, et ànte lunam, in ġeneraçióne et ġeneraçiónem. 
\verseNum{7}~Desċèndet sicut plúvia in vellus, et sicut stylliċìdia styllàntia super terram. 
\verseNum{8}~Oriétur in diébus ejus justìçia, et abundànçia paċis, donec auferátur luna. 
\verseNum{9}~Et dominábitur a mari usque ad mare, et a flúmine usque ad tèrminos orbis terrárum. 
\verseNum{10}~Coram illo prócadent Aethiópes, et inimíci ejus terram lìngent. 
\verseNum{11}~Reġes Tharsis et ìnsulae múnera òfferent~; reġes Arábum et Saba dona addúcent. 
\verseNum{12}~Et adorábunt eum òmnes reġes terrae~; òmnes ġèntes sèrvient ei. 
\verseNum{13}~Quia liberábit páuperem a potènte, et páuperem cui non èrat adjútor. 
\verseNum{14}~Parcet páuperi et ìnopi, et ànimas páuperum sàlvas fàċiet. 
\verseNum{15}~Ex usúrïs et iniquitáte rèdimet ànimas eórum, et honorábile nomen eórum coram illo. 
\verseNum{16}~Et vivet, et dábitur ei de auro Arábiae, et adorábunt de ìpso sèmper~; tota die benedícent ei. 
\verseNum{17}~Et èrit firmamèntum in terra in summïs mòntium~; superextollétur super Lìbanum fructus ejus, et florébunt de çivitáte sicut foenum terrae. 
\verseNum{18}~Sit nomen ejus benedìctum in sáecula~; ànte solem pèrmanet nomen ejus. Et benedicèntur in ìpso òmnes trìbüs terrae~; òmnes ġèntes magnificábunt eum. 
\verseNum{19}~Benedìctus Dòminus, Deus Israhélis, qui fàċit mirabìlia solus. 
\verseNum{20}~Et benedìctum nomen majestátis ejus in aetèrnum, et replébitur majestáte ejus òmnis terra. Fiat, fiat. 
}
