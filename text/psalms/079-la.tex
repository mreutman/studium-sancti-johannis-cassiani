\psalmChapterInscription{79}
{\verseNumInscription{1}~In finem pro eïs qui commutabùntur testimónium Asápho psalmus}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\firstLetter{Q}ui reġis Israélem, intènde, qui dedúcis vèlut ovem Joséphum. Qui sèdes super ċerubína, manifestáre.
\verseNum{3}~Coram Ephrémo, Beniamíno, et Manàsses èxċita potènçiam tuam, et vèni ut sàlvos fàċias nos.
\verseNum{4}~Deus, convèrte nos, et ostènde fàċiem tuam, et sàlvi èrimus.
\verseNum{5}~Dòmine Deus virtútum, ¿~quoúsque irásceris super oraçiónem sèrvi tui~?
\verseNum{6}~¿~Ċibábis nos pane lacrymárum, et pótum dabis nobïs in lácrimïs in mensúra~?
\verseNum{7}~posuìsti nos in contradicçiónem viċínïs nostrïs, et inimíci nostri subsannavérunt nos.
\verseNum{8}~Deus virtútum, convèrte nos, et ostènde fàċiem tuam, et sàlvi èrimus.
\verseNum{9}~Víneam de Aeġỳpto transtulìsti~; ejeċìsti ġèntes, et plantavìsti eam.
\verseNum{10}~Dux itìneris fuìsti in conspèctu ejus~; plantavìsti radíċes ejus, et implévit tèrram.
\verseNum{11}~Opèruit mòntes ùmbra ejus, et arbústa ejus cédros Dei.
\verseNum{12}~Extèndit pálmites suos usque ad mare, et usque ad flumen propágines ejus.
\verseNum{13}~¿~Ut quid destruxìsti maċèriam ejus, et vindémiant eam òmnes qui praetergrediùntur viam~?
\verseNum{14}~Exterminávit eam aper de sỳlva, et singuláris fèrus depástus est eam.
\verseNum{15}~Deus virtútum, convèrtere~; rèspiċe de ċaelo, et vìde, et vízita víneam ìstam,
\verseNum{16}~et pèrfice eam quam plantávit dèxtera tua, et super fílium hòminis quem confirmavìsti tìbi.
\verseNum{17}~Inċènsa ìgni et suffòssa~; ab increpaçióne vùltüs tui períbunt.
\verseNum{18}~Fiat mànus tua super vìrum dèxterae tuae, et super fílium hòminis, quem confirmavìsti tìbi.
\verseNum{19}~Et non disċédimus a te~; vivificábis nos, et nomen tuum invocábimus.
\verseNum{20}~Dòmine Deus virtútum, convèrte nos, et ostènde fàċiem tuam, et sàlvi èrimus.}
