\psalmChapterWithInscription{}
{\verseNumInscription{1} }
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{D}iléxi, quóniam exáudiet Dòminus voċem oraçiónis meae. 
\verseNum{2}~Quia inclinávit aurem suam mìhi, et in diébus meïs invocábo. 
\verseNum{3}~Çircumdedérunt me dolóres mortis, et perícula infèrni invenérunt me~; tribulaçiónem et dolórem invóni, 
\verseNum{4}~et nomen Dòmini invocávi~; O Dòmine, líbera ànimam meam. 
\verseNum{5}~Mizèricors Dòminus, et justus, et Deus noster mizerétur. 
\verseNum{6}~Custódiens párvulos Dòminus~; humiliátus sum, et liberávit me. 
\verseNum{7}~Convèrtere, ànima mea, in rèquiem tuam, quia Dòminus beneféċit tìbi. 
\verseNum{8}~Quia erìpuit ànimam meam de morte, òculos meos a làcrymïs, pèdes meos a lápsu. 
\verseNum{9}~Plaċébo Dòmino in reġióne vivórum. 
}
