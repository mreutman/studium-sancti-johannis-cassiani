\psalmChapterInscription{98}
{\verseNumInscription{1}~Psalmus ìpsi Davídis}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{D}òminus regnávit~; irascàntur pòpuli. Qui sèdet super ċerubíma~; moveátur terra.
\verseNum{2}~Dòminus in Sióne magnus, et exċèlsus super òmnes pòpulos.
\verseNum{3}~Confiteàntur nómini tuo magno, quóniam terrìbile et sanctum est.
\verseNum{4}~Et honor reġis judìċium díligit~; tu paravìsti direcçiónes~; judìċium et justìçiam in Jacóbe tu feċìsti.
\verseNum{5}~Exaltáte Dòminum Deum nostrum, et adoráte scabèllum pèdum ejus, quóniam sanctum est.
\verseNum{6}~Moýses et Áharon in saċerdótibus ejus, et Sámuel inter eos qui ìnvocant nomen ejus. Invocábant Dòminum, et ìpse exaudiébat eos.
\verseNum{7}~In colùmna nubis loquebátur ad eos~; custodiébant testimónia ejus, et praeċèptum quod dèdit illïs.
\verseNum{8}~Dòmine Deus noster, tu exaudiébas eos~; Deus tu propìçius fuìsti eïs, et ulċíscens in òmnes adinvençiónes eórum.
\verseNum{9}~Exaltáte Dòminum Deum nostrum, et adoráte in mònte sancto ejus, quóniam sanctus Dòminus Deus noster.
}
