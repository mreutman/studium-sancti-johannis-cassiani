\psalmChapterInscription{150}
{\verseNumInscription{1}~Allelúja}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\firstLetter{L}audáte Dòminum in sanctīs ejus~; laudáte eum in firmamènto virtútis ejus.
\verseNum{2}~Laudáte eum in virtútibus ejus~; laudáte eum secùndum multitúdinem magnitúdinis ejus.
\verseNum{3}~Laudáte eum in sono tubae~; laudáte eum in psaltèrio et çìtara.
\verseNum{4}~Laudáte eum in tỳmpano et choro~; laudáte eum in chordīs et órgano.
\verseNum{5}~Laudáte eum in ċỳmbalīs bène sonàntibus~; laudáte eum in ċỳmbalīs jubilaçiónis.
\verseNum{6}~Òmnis spíritus laudet Dòminum. Allelúja.}
