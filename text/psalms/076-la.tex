\psalmChapterWithInscription{76}
{\verseNumInscription{1} In finem pro Idithúno psalmus Asápho}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\verseFirstLetter{V}oċe mea ad Dòminum clamávi~; voċe mea ad Deum, et intèndit mìhi. 
\verseNum{3}~In die tribulaçiónis meae Deum exquisívi, mànibus meïs nòcte còntra eum, et non sum deċèptus~; renúvit consolári ànima mea. 
\verseNum{4}~Mèmor fui Dei, et delectátus sum, et exerċitátus sum, et deféċit spíritus meus. 
\verseNum{5}~Antiċipavérunt viġìlias òculi mei~; turbátus sum, et non sum locútus. 
\verseNum{6}~Coġitávi dies antíquos, et annos aetèrnos in mente hábui, 
\verseNum{7}~Et meditátus sum nòcte cum corde meo, et exerċitábar, et inspiċiébam spíritum meum. 
\verseNum{8}~¿~Numquid in aetèrnum projìċiet Deus, et non appónet ut complaċìtior sit adhüc~? 
\verseNum{9}~¿~Aut in finem mizericórdiam suam absċìndet, a ġeneraçióne in ġeneraçiónem~? 
\verseNum{10}~¿~Aut obliviscétur mizeréri Deus~? ¿~Aut continébit in ira sua mizericórdias suas~? 
\verseNum{11}~Et dixi, «~Nunc coepi~; haec mutáçio dèxterae Exċèlsi.~»
\verseNum{12}~Mèmor fui óperum Dòmini, quia mèmor èro ab inìçio mirabílium tuórum, 
\verseNum{13}~et meditábor in òmnibus opèribus tuïs, et in adinvençiónibus tuïs exerċébor. 
\verseNum{14}~Deus, in sancto via tua~; ¿~quìs Deus magnus sicut Deus noster~? 
\verseNum{15}~Tu es Deus qui fàċis mirabìlia~; notam feċìsti in pòpulïs virtútem tuam. 
\verseNum{16}~Redemìsti in bráchio tuo pòpulum tuum, fílios Jacóbi et Joséphi. 
\verseNum{17}~Vïdérunt te aquae, Deus~; vïdérunt te aquae, et timuérunt, et turbátae sunt abỳssi. 
\verseNum{18}~Multitúdo sonátüs aquárum~; voċem dedérunt nubes, ètenim saġìttae tuae trànseunt. 
\verseNum{19}~Vox tonìtrui tui in rota~; illuxérunt coruscaçiónes tuae orbi terrae~; commóta est, et contrèmuit terra. 
\verseNum{20}~In mari via tua, et sémitae tuae in aquïs mùltïs, et vestíġia tua non cognoscèntur. 
\verseNum{21}~Deduxìsti sicut oves pòpulum tuum, in mànu Moýsis et Aharónis. 
}
