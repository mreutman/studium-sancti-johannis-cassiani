\psalmChapterInscription{76}
{\verseNumInscription{1}~In finem pro Jedutúne psàlmus Asáfo}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\firstLetter{V}oċe mea ad Dòminum clamávi~; voċe mea ad Deum, et intèndit mìhi.
\verseNum{3}~In die tribulaçiónis meae Deum exquisívi, mànibus meīs nòcte còntra eum, et non sum deċèptus~; renúvit consolári ànima mea.
\verseNum{4}~Mèmor fui Dei, et delectátus sum, et exerċitátus sum, et deféċit spíritus meus.
\verseNum{5}~Antiċipavérunt viġìlias òculi mei~; turbátus sum, et non sum locútus.
\verseNum{6}~Coġitávi dies antíquos, et annos aetèrnos in mente hábui,
\verseNum{7}~Et meditátus sum nòcte cum corde meo, et exerċitábar, et inspiċiébam spíritum meum.
\verseNum{8}~¿~Numquid in aetèrnum projìċiet Deus, et non appónet ut complaċìtior sit adhūc~?
\verseNum{9}~¿~Aut in finem mizericórdiam suam absċìndet, a ġeneraçióne in ġeneraçiónem~?
\verseNum{10}~¿~Aut obliviscétur mizeréri Deus~? ¿~Aut continébit in ira sua mizericórdias suas~?
\verseNum{11}~Et dixi, «~Nunc coepi~; haec mutáçio dèxterae Exċèlsi.~»
\verseNum{12}~Mèmor fui óperum Dòmini, quia mèmor èro ab inìçio mirabílium tuórum,
\verseNum{13}~et meditábor in òmnibus opèribus tuīs, et in adinvençiónibus tuīs exerċébor.
\verseNum{14}~Deus, in sancto via tua~; ¿~quìs Deus màgnus sicut Deus noster~?
\verseNum{15}~Tu es Deus qui fàċis mirabìlia~; notam feċìsti in pòpulīs virtútem tuam.
\verseNum{16}~Redemìsti in bráchio tuo pòpulum tuum, fílios Jacóbi et Joséfi.
\verseNum{17}~Vīdérunt te aquae, Deus~; vīdérunt te aquae, et timuérunt, et turbátae sunt abỳssi.
\verseNum{18}~Multitúdo sonátūs aquárum~; voċem dedérunt nubes, ètenim saġìttae tuae trànseunt.
\verseNum{19}~Vox tonìtrui tui in rota~; illuxérunt coruscaçiónes tuae orbi tèrrae~; commóta est, et contrèmuit tèrra.
\verseNum{20}~In mari via tua, et sémitae tuae in aquīs mùltīs, et vestíġia tua non cognoscèntur.
\verseNum{21}~Deduxìsti sicut oves pòpulum tuum, in mànu Moýsis et Aharónis.}
