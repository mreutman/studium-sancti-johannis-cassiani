\psalmChapterInscription{50}
{\verseNumInscription{1}~In finem psalmus Davídi \verseNumInscription{2}~quum vēnit ad eum Nathánus proféta quando intrávit ad Betsábeam }
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{3}~\firstLetter{M}izerére mei, Deus, secùndum màgnam mizericórdiam tuam, et secùndum multitúdinem mizeraçiónum tuárum dele iniquitátem meam.
\verseNum{4}~Àmplius lava me ab iniquitáte mea, et a peccáto meo mùnda me,
\verseNum{5}~quóniam iniquitátem meam ego cognósco, et peccátum meum còntra me est sèmper.
\verseNum{6}~Tìbi soli peccávi, et màlum coram te feċi, ut justificéris in sermónibus tuīs, et vincas quum judicáris.
\verseNum{7}~Ècċe ènim in iniquitátibus conċèptus sum, et in peccátīs conċépit me mater mea.
\verseNum{8}~Ècċe ènim veritátem dilexìsti~; inçèrta et occùlta sapiènçiae tuae manifestavìsti mìhi.
\verseNum{9}~Aspèrges me hyssópo, et mundábor~; lavábis me, et super nivem dealbábor.
\verseNum{10}~Audítui meo dabis gáudium et laetìçiam, et exsultábunt òssua humiliáta.
\verseNum{11}~Avèrte fàċiem tuam a peccátīs meīs, et òmnes iniquitátes meas dele.
\verseNum{12}~Cor mùndum crea in me, Deus, et spíritum rectum ìnnova in viscèribus meīs.
\verseNum{13}~Ne projìċias me a fàċie tua, et spíritum sanctum tuum ne áuferas a me.
\verseNum{14}~Redde mìhi laetìçiam salutáris tui, et spíritu prinċipáli confìrma me.
\verseNum{15}~Doċébo iníquos vias tuas, et ìmpii ad te convertèntur.
\verseNum{16}~Líbera me de sangüìnibus, Deus, Deus salútis meae, et exsultábit lìngüa mea justìçiam tuam.
\verseNum{17}~Dòmine, lábia mea apèries, et os meum adnunçiábit laudem tuam.
\verseNum{18}~Quóniam si voluìsses sacrifìċium, dedìssem útique~; holocáustis non delectáberis.
\verseNum{19}~Sacrifìċium Deo spíritus contribulátus~; cor contrítum et humiliátum, Deus, non despìċies.
\verseNum{20}~Benígne fac, Dòmine, in bona voluntáte tua Ziónem, ut aedificèntur muri Hierozólymae~;
\verseNum{21}~tunc acċeptábis sacrifìċium justìçiae, oblaçiónes et holocáusta~; tunc impónent super altáre tuum vìtulos.}
