\psalmChapterInscription{82}
{\verseNumInscription{1}~Cànticum psalmi Asápho}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\verseFirstLetter{D}eus, ¿~quìs sìmilis èrit tìbi~? Ne táċeas, néque compescáris, Deus, 
\verseNum{3}~quóniam ecċe inimíci tui sonavérunt, et qui odérunt te extulérunt caput. 
\verseNum{4}~Super pòpulum tuum malignavérunt consìlium, et coġitavérunt advèrsus sanctos tuos. 
\verseNum{5}~Dixérunt, «~Veníte, et disperdámus eos de ġènte, et non memorétur nomen Israhélis ùltra.~» 
\verseNum{6}~Quóniam coġitavérunt unanìmiter~; sìmul advèrsum te testamèntum disposuérunt, 
\verseNum{7}~tabernácula Idumaeórum et Ismáelitum, Mohábis et Hagaréni, 
\verseNum{8}~Gebális et Ammóni et Amaléci, alieníġenae cum habitàntibus Tyri, 
\verseNum{9}~ètenim Assur vënit cum illïs, fàcti sunt in adjutórium fíliïs Lòtis. 
\verseNum{10}~Fac illïs sicut Midiáni et Síserae, sicut Jabíni in torrènte Ċisónis. 
\verseNum{11}~Disperivérunt in Endóre~; fàcti sunt ut stercus terrae. 
\verseNum{12}~Pone prínċipes eórum sicut Orébem et Sébem, et Sebáhem et Zalmúnam~; òmnes prínċipes eórum, 
\verseNum{13}~qui dixérunt, «~Hereditáte possideámus sanctuárium Dei.~»
\verseNum{14}~Deus meus, pone illos ut rotam, et sicut stìpulam ànte fàċiem vènti. 
\verseNum{15}~sicut ìgnis qui combúrit sỳlvam, et sicut flamma combúrens mòntes, 
\verseNum{16}~ita persèqueris illos in tempestáte tua, et in ira tua turbábis eos. 
\verseNum{17}~Imple fàċies eórum ignomìnia, et quaerent nomen tuum, Dòmine. 
\verseNum{18}~Erubéscant, et conturbèntur in sáeculum sáeculi, et confundàntur, et pèreant. 
\verseNum{19}~Et cognóscant quia nomen tìbi Dòminus~; tu solus Altìssimus in òmni terra. 
}
