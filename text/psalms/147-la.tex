\psalmChapterWithInscription{147}
{\verseNumInscription{1} Allelúja}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{L}auda, Jerozólyma, Dòminum~; lauda Deum tuum, Sion. 
\verseNum{2}~Quóniam confortávit sèras portárum tuárum~; benedíxit fíliïs tuïs in te. 
\verseNum{3}~Qui pósuit fines tuos paċem, et àdipe frumènti sátiat te. 
\verseNum{4}~Qui emìttit elóquium suum terrae~; velóċiter currit sermo ejus. 
\verseNum{5}~Qui dat nivem sicut lanam~; nèbulam sicut ċìnerem spargit. 
\verseNum{6}~Mittit crystàllum suum sicut buccèllas~; ¿~ànte fàċiem frígoris ejus quìs sustinébit~? 
\verseNum{7}~Emìttet vèrbum suum, et liquefáċiet ea~; flabit spíritus ejus, et fluent aquae. 
\verseNum{8}~Qui adnúnçiat vèrbum suum Jacóbi~; justìçias et judìċia sua Izrahéli. 
\verseNum{9}~Non feċit táliter òmni naçióni, et judìċia sua non manifestávit eïs. 
}
