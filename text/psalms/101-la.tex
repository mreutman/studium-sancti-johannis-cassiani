\psalmChapterWithInscription{101}
{\verseNumInscription{1} Oráçio páuperis quum ánxius fúerit et in conspèctu dòmini effúderit prèċem suam}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\verseFirstLetter{D}òmine exáudi oraçiónem meam, et clamor meus ad te vèniat. 
\verseNum{3}~Non avèrtas fàċiem tuam a me. In quacùmque die trìbulor, inclína ad me aurem tuam~; in quacùmque die invocávero te, velóċiter exáudi me. 
\verseNum{4}~Quia defeċérunt sicut fumus dies mei, et òssua mea sicut crèmium aruérunt. 
\verseNum{5}~Percùssus sum ut foenum, et áruit cor meum, quia oblítus sum comèdere panem meum. 
\verseNum{6}~A voċe ġèmitüs mei adháesit òssu meum càrni meae. 
\verseNum{7}~Sìmilis fàctus sum pelicáno solitúdinis~; fàctus sum sicut noctỳcorax in domicìlio. 
\verseNum{8}~Viġilávi, et fàctus sum sicut passer solitárius in tecto. 
\verseNum{9}~Tota die exprobrábant mìhi inimíci mei, et qui laudábant me advèrsum me jurábant~; 
\verseNum{10}~quia ċìnerem tàmquam panem manducábam, et potum meum cum flétu misċébam, 
\verseNum{11}~a fàċie irae et indignaçiónis tuae~; quia élevans allisìsti me. 
\verseNum{12}~Dies mei sicut ùmbra declinavérunt, et ego sicut foenum árui. 
\verseNum{13}~Tu autem, Dòmine, in aetèrnum pèrmanes, et memoriále tuum in ġeneraçiónem et ġeneraçiónem. 
\verseNum{14}~Tu exsùrgens mizeréberis Siónis, quia tempus mizerèndi ejus, quia vënit tempus. 
\verseNum{15}~Quóniam plaċuérunt sèrvïs tuïs lápides ejus, et terrae ejus mizerebùntur. 
\verseNum{16}~Et timébunt ġèntes nomen tuum, Dòmine, et òmnes reġes terrae glóriam tuam. 
\verseNum{17}~Quia aedificávit Dòminus Siónem, et vidébitur in glória sua. 
\verseNum{18}~Respèxit in oraçiónem humìlium et non sprevit prèċem eórum. 
\verseNum{19}~Scribàntur haec in ġeneraçióne àltera, et pòpulus qui creábitur laudábit Dòminum. 
\verseNum{20}~Quia prospèxit de exċèlso sancto suo~; Dòminus de ċaelo in terram aspèxit, 
\verseNum{21}~ut audíret ġèmitüs compeditórum~; ut sòlveret fílios interemptórum~; 
\verseNum{22}~ut adnúnçiet in Sióne nomen Dòmini, et laudem ejus in Jerozólyma. 
\verseNum{23}~In convenièndo pòpulos in unum, et reġes ut sèrviant Dòmino. 
\verseNum{24}~Respòndit ei in via virtútis suae, «~Pauċitátem diérum meórum núnçia mìhi.
\verseNum{25}~Ne rèvoċes me in dimìdio diérum meórum, in ġeneraçiónem et ġeneraçiónem anni tui.~»
\verseNum{26}~Inìçio tu, Dòmine, terram fundavìsti, et ópera mànuum tuárum sunt ċaeli. 
\verseNum{27}~Ìpsi períbunt, tu autem pèrmanes, et òmnes sicut vestimèntum veteráscent, et sicut opertórium mutábis eos, et mutabùntur~; 
\verseNum{28}~tu autem ídem ìpse es, et anni tui non defìċient. 
\verseNum{29}~Fílii servórum tuórum habitábunt, et semen eórum in sáeculum dirigétur. 
}
