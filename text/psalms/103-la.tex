\psalmChapterInscriptionBeginDivision{103}
{\verseNumInscription{1}~Ìpsi Davídis}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{B}enedíce, ànima mea, Dòmino. Dòmine Deus meus, magnificátus es vehemènter. Confessiónem et decórem induìsti,
\verseNum{2}~amìctus lúmine sicut vestimènto, extèndens ċaelum sicut pellem.
\verseNum{3}~Qui tègis aquïs superióra ejus~; qui ponis nubem asċènsum tuum~; qui àmbulas super pennas ventórum.
\verseNum{4}~Qui fàċis ànġelos tuos spíritüs, et minìstros tuos ìgnem urèntem.
\verseNum{5}~Qui fundavìsti terram super stabilitátem suam~; non inclinábitur in sáeculum sáeculi.
\verseNum{6}~Abỳssus sicut vestimèntum amìctus ejus~; super mòntes stabunt aquae.
\verseNum{7}~Ab increpaçióne tua fúġient~; a voċe tonìtrui tui formidábunt.
\verseNum{8}~Asċèndunt mòntes, et desċèndunt campi in locum quem fundavìsti eïs.
\verseNum{9}~Tèrminum posuìsti quem non transgredièntur, néque convertèntur operíre terram.
\verseNum{10}~Qui emìttïs fòntes in convàllibus~; inter mèdium mòntium pertransíbunt aquae.
\verseNum{11}~Potábunt òmnes béstiae agri~; expectábunt ònagri in sìti sua.
\verseNum{12}~Super ea vulácres ċaeli habitábunt~; de mèdio petrárum dabunt voċem.
\verseNum{13}~Rìgans mòntes de superióribus suïs~; de fructu óperum tuórum satiábitur terra~;
\verseNum{14}~prodúcens foenum jumèntïs, et herbam servitúti hòminum, ut edúcas panem de terra,
\verseNum{15}~et vínum laetìficet cor hòminis, ut exhìlaret fàċiem in óleo, et panis cor hòminis confìrmet.
\verseNum{16}~Saturabùntur lìgna campi, et cédri Lìbani quas plantávit~;
\verseNum{17}~illic pàsseres nidificábunt~; heródii domus dux est eórum.
\verseNum{18}~Mòntes exċèlsi ċèrvïs~; pètra refúġium herináċeïs.
\verseNum{19}~Feċit lunam in tèmpora~; sol cognóvit occásum suum.
\verseNum{20}~posuìsti tènebras, et fàcta est nòx~; in ìpsa pertransíbunt òmnes béstiae sỳlvae.
\verseNum{21}~cátuli leónum ruġièntes ut rápiant, et quaerant a Deo escam sìbi.
\verseNum{22}~Ortus est sol, et congregáti sunt, et in cubílibus suïs collocabùntur.
\verseNum{23}~Exíbit hòmo ad opus suum, et ad operaçiónem suam usque ad vèsperum.
\verseNum{24}~Quam magnificáta sunt ópera tua, Dòmine. Òmnia in sapiènçia feċìsti~; impléta est terra possessióne tua.
\verseNum{25}~Hoc mare magnum, et spaçiósum mànibus~; illic reptìlia quórum non est númerus, animália pusìlla cum magnïs.
\verseNum{26}~Illïc naves pertransíbunt~; draco ìste quem formavìsti ad illudèndum ei.
\verseNum{27}~Òmnia a te exspèctant ut des illïs escam in tèmpore~;
\verseNum{28}~dante te illïs, còllegent~; aperiènte te mànum tuam, òmnia implebùntur bonitáte.
\verseNum{29}~Avertènte autem te fàċiem, turbabùntur~; áuferes spíritum eórum, et defìċient, et in pùlverem suum revertèntur.
\verseNum{30}~Emìttes spíritum tuum, et creabùntur, et renovábis fàċiem terrae.
\verseNum{31}~Sit glória Dòmini in sáeculum~; laetábitur Dòminus in opèribus suïs.
\verseNum{32}~Qui rèspiċit terram, et fàċit eam trèmere~; qui tangit mòntes, et fúmigant.
\verseNum{33}~Cantábo Dòmino in vita mea~; psallem Deo meo quámdiu sum.
\verseNum{34}~Jucùndum sit ei elóquium meum~; ego vero delectábor in Dòmino.
\verseNum{35}~defìċiant peccatóres a terra, et iníqui, ita ut non sint. Benedíce, ànima mea, Dòmino.}
