\psalmChapterInscriptionBeginDivision{105}
{\verseNumInscription{1}~Allelúja}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\firstLetter{C}onfitémini Dòmino quóniam bonus, quóniam in sáeculum mizericórdia ejus.
\verseNum{2}~¿~Quìs loquétur potènçias Dòmini, audítas fàċiet òmnes laudes ejus~?
\verseNum{3}~Beáti qui custódiunt judìċium, et fàċiunt justìçiam in òmni tèmpore.
\verseNum{4}~Memènto nostri, Dòmine, in benepláċito pòpuli tui~; vízita nos in salutári tuo,
\verseNum{5}~ad vidèndum in bonitáte electórum tuórum, ad laetandum in laetìçia ġentis tuae, ut laudéris cum hereditáte tua.
\verseNum{6}~Peccávimus cum pàtribus nostrīs~; injúste égimus~; iniquitátem féċimus.
\verseNum{7}~Patres nostri in Aeġỳpto non intellēxérunt mirabìlia tua~; non fuérunt mèmores multitúdinis mizericórdiae tuae, et irritavérunt asċendèntes in mare, mare Rubrum.
\verseNum{8}~Et salvávit eos pròpter nomen suum, ut notam fàċeret potènçiam suam.
\verseNum{9}~Et increpávit mare Rubrum et exsiccátum est, et dedúxit eos in abỳssīs sicut in desèrto.
\verseNum{10}~Et salvávit eos de mànu odièntium, et redémit eos de mànu inimíci.
\verseNum{11}~Et opèruit aqua tribulàntes eos~; unus ex eīs non remánsit.
\verseNum{12}~Et credidérunt vèrbīs ejus, et laudavérunt laudem ejus.
\verseNum{13}~Ċite feċérunt~; oblíti sunt óperum ejus, et non sustinuérunt consìlium ejus.
\verseNum{14}~Et concupivérunt concupiscènçiam in desèrto, et tentavérunt Deum in inaquóso.
\verseNum{15}~Et dèdit eīs petiçiónem ipsórum, et misit saturitátem in ànimas eórum.
\verseNum{16}~Et irritavérunt Moýsem in castrīs~; Aharónem, sanctum Dòmini.
\verseNum{17}~Apèrta est tèrra, et degluttívit Dathánum, et opèruit super congregaçiónem Abirónis.
\verseNum{18}~Et exársit ìgnis in synagóga eórum~; flamma combùssit peccatóres.
\verseNum{19}~Et feċérunt vìtulum in Horébe, et adoravérunt scùlptile.
\verseNum{20}~Et mutavérunt glóriam suam in similitúdinem vìtuli comedèntis foenum.
\verseNum{21}~Oblíti sunt Deum qui salvávit eos~; qui feċit magnália in Aeġỳpto,
\verseNum{22}~mirabìlia in tèrra Ċami, terribìlia in mari Rubro.
\verseNum{23}~Et dixit ut dispèrderet eos, si non Moýses eléctus ejus stetìsset in confracçióne in conspèctu ejus, ut avèrteret iram ejus ne dispèrderet eos.
\verseNum{24}~Et pro nìhilo habuérunt tèrram desiderábilem~; non credidérunt vèrbo ejus.
\verseNum{25}~Et murmuravérunt in tabernáculīs suīs~; non exaudivérunt voċem Dòmini.
\verseNum{26}~Et elevávit mànum suam super eos, ut prostèrneret eos in desèrto.
\verseNum{27}~Et ut dejìċeret semen eórum in naçiónibus, et dispèrgeret eos in reġiónibus.
\verseNum{28}~Et iniçiáti sunt Beelfegóri, et comedérunt sacrifìċia mortuórum.
\verseNum{29}~Et irritavérunt eum in adinvençiónibus suīs, et multiplicáta est in eīs ruína.
\verseNum{30}~Et stètit Fìneas et placávit, et ċessávit quassáçio.
\verseNum{31}~Et reputátum est ei in justìçiam, in ġeneraçiónem et ġeneraçiónem ùsque in sempitèrnum.
\verseNum{32}~Et irritavérunt eum ad aquas contradicçiónis, et vexátus est Moýses pròpter eos,
\verseNum{33}~quia exaċerbavérunt spíritum ejus, et distínxit in lábiīs suīs.
\verseNum{34}~Non disperdidérunt ġèntes quas dixit Dòminus ìllīs.
\verseNum{35}~Et commìxti sunt inter ġèntes, et didicérunt ópera eórum,
\verseNum{36}~et servivérunt sculptìlibus eórum, et fàctum est ìllīs in scàndalum.
\verseNum{37}~Et immolavérunt fílios suos, et fílias suas daemòniīs.
\verseNum{38}~Et effudérunt sàngüinem innoċèntem, sàngüinem filiórum suórum et filiárum suárum quas sacrificavérunt sculptìlibus Canaánis, et infècta est tèrra in sangüìnibus.
\verseNum{39}~Et contamináta est in opèribus eórum, et fornicáti sunt in adinvençiónibus suīs.
\verseNum{40}~Et irátus est furóre Dòminus in pòpulum suum, et abominátus est hereditátem suam.
\verseNum{41}~Et trádidit eos in mànūs ġèntium, et domináti sunt eórum qui odérunt eos.
\verseNum{42}~Et tribulavérunt eos inimíci eórum, et humiliáti sunt sub mànibus eórum.
\verseNum{43}~Sáepe liberávit eos~; ìpsi autem exaċerbavérunt eum in consìlio suo, et humiliáti sunt in iniquitátibus suīs.
\verseNum{44}~Et vīdit quum tribularèntur, et audívit oraçiónem eórum.
\verseNum{45}~Et mèmor fuit testamènti sui, et poenìtuit eum secùndum multitúdinem mizericórdiae suae.
\verseNum{46}~Et dèdit eos in mizericórdias in conspèctu òmnium qui céperant eos.
\verseNum{47}~Sàlvos fac nos, Dòmine Deus noster, et còngrega nos de naçiónibus, ut confiteámur nómini sancto tuo, et gloriémur in laude tua.
\verseNum{48}~Benedìctus Dòminus Deus Israélis, a sáeculo et ùsque in sáeculum. Et dicet òmnis pòpulus, «~Fiat, fiat.~»}
