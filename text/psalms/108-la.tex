\psalmChapterInscription{108}
{\verseNumInscription{1}~In finem psalmus Davídis}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\firstLetter{D}eus, laudem meam ne tacúerïs, quia os peccatóris et os dolósi super me apèrtum est.
\verseNum{3}~Locúti sunt advèrsum me lìngüa dolósa, et sermónibus ódii çircumdedérunt me, et expugnavérunt me gratïs.
\verseNum{4}~Pro eo ut me dilìgerent, detrahébant mìhi~; ego autem orábam.
\verseNum{5}~Et posuérunt advèrsum me màla pro bonïs, et ódium pro delecçióne mea.
\verseNum{6}~Constìtue super eum peccatórem, et diábolus stet a dèxtrïs ejus.
\verseNum{7}~Quum judicátur, èxiat condemnátus, et oráçio ejus fiat in peccátum.
\verseNum{8}~Fiant dies ejus pauci, et episcopátum ejus acċìpiat alter.
\verseNum{9}~Fiant fílii ejus órphani, et uxor ejus vìdua.
\verseNum{10}~Nutàntes transferàntur fílii ejus et mendíċent, et ejiċiàntur de habitaçiónibus suïs.
\verseNum{11}~Scrutétur foenerátor òmnem substànçiam ejus, et dirìpiant aliéni labóres ejus.
\verseNum{12}~Non sit illi adjútor, nec sit qui mizereátur pupìllïs ejus.
\verseNum{13}~Fiant nati ejus in intèritum~; in ġeneraçióne una deleátur nomen ejus.
\verseNum{14}~In memóriam rèdiat iníquitas pàtrum ejus in conspèctu Dòmini, et peccátum matris ejus non deleátur.
\verseNum{15}~Fiant còntra Dòminum sèmper, et dispèreat de terra memória eórum,
\verseNum{16}~pro eo quod non est recordátus fàċere mizericórdiam, et persecútus est hòminem ìnopem et mendícum, et compúnctum corde, mortificáre.
\verseNum{17}~Et diléxit maledicçiónem, et vèniet ei. Et nóluit benedicçiónem, et elongábitur ab eo.
\verseNum{18}~Et indúvit maledicçiónem sicut vestimèntum, et intrávit sicut aqua in interióra ejus, et sicut óleum in òssibus ejus.
\verseNum{19}~Fiat ei sicut vestimèntum, quo operítur, et sicut zona qua sèmper praeċìngitur.
\verseNum{20}~Hoc opus eórum qui détrahunt mìhi àpud Dòminum, et qui loquùntur màla advèrsus ànimam meam.
\verseNum{21}~Et tu, Dòmine, Dòmine, fac mecum pròpter nomen tuum, quia süávis est mizericórdia tua.
\verseNum{22}~Líbera me, quia eġénus et pauper ego sum, et cor meum conturbátum est intra me.
\verseNum{23}~sicut ùmbra quum declínat ablátus sum, et excùssus sum sicut locústae.
\verseNum{24}~Ġènua mea infirmáta sunt ab jejúnio, et càro mea immutáta est pròpter óleum.
\verseNum{25}~Et ego fàctus sum oppróbrium illïs~; vïdérunt me, et movérunt cápita sua.
\verseNum{26}~Àdjuva me, Dòmine Deus meus~; sàlvum me fac secùndum mizericórdiam tuam.
\verseNum{27}~Et sciant quia mànus tua haec, et tu, Dòmine, feċìsti eam.
\verseNum{28}~Maledícent illi, et tu benedíces~; qui insùrgunt in me, confundàntur, sèrvus autem tuus laetábitur.
\verseNum{29}~Induàntur qui détrahunt mìhi pudóre, et operiàntur sicut mantéli confuzióne sua.
\verseNum{30}~Confitébor Dòmino nìmis in ore meo, et in mèdio multórum laudábo eum.
\verseNum{31}~Quia àstitit a dèxtrïs páuperis, ut sàlvam fàċeret a persequèntibus ànimam meam.}
