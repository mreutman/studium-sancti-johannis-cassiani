\psalmChapterWithInscription{104}
{\verseNumInscription{1} Allelúja}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{C}onfitémini Dòmino, et invocáte nomen ejus~; adnunçiáte inter ġèntes ópera ejus. 
\verseNum{2}~Cantáte ei, et psàllite ei~; narráte òmnia mirabìlia ejus. 
\verseNum{3}~Laudámini in nómine sancto ejus~; laetétur cor quaerèntium Dòminum. 
\verseNum{4}~Quaerite Dòminum, et confirmámini~; quáerite fàċiem ejus sèmper. 
\verseNum{5}~Mementóte mirabílium ejus quae feċit~; prodìġia ejus, et judìċia oris ejus. 
\verseNum{6}~Semen Abrahámi, sèrvi ejus~; fílii Jacóbi, elécti ejus. 
\verseNum{7}~Ìpse Dòminus Deus noster~; in univèrsa terra judìċia ejus. 
\verseNum{8}~Mèmor fuit in sáeculum testamenti sui~; vèrbi quod mandávit in mille ġeneraçiónes, 
\verseNum{9}~quod dispósuit ad Abrahámum, et juramènti sui ad Isácum. 
\verseNum{10}~Et statúvit illud Jacóbi in praeċèptum, et Israhéli in testamèntum aetèrnum, 
\verseNum{11}~dicens, «~Tìbi dabo terram Canaánis, funìculum hereditátis vèstrae.~» 
\verseNum{12}~Quum essent número brèvi, paucìssimi, et ìncolae ejus, 
\verseNum{13}~et pertransivérunt de ġènte in ġèntem, et de regno ad pòpulum àlterum. 
\verseNum{14}~Non relíquit hòminem noċére eïs, et corrìpuit pro eïs reġes. 
\verseNum{15}~Nolíte tángere christos meos, et in prophétïs meïs nolíte malignári. 
\verseNum{16}~Et vocávit famem super terram, et omne firmamèntum panis contrívit. 
\verseNum{17}~misit ànte eos vìrum~; in servum venundátus est Joséphus. 
\verseNum{18}~Humiliavérunt in compèdibus pèdes ejus, ferrum pertransívit ànimam ejus, 
\verseNum{19}~donec věníret vèrbum ejus~; elóquium Dòmini inflammávit eum. 
\verseNum{20}~misit rex et solvit eum, prinċeps populórum, et dímisit eum. 
\verseNum{21}~Constitúvit eum dòminum domüs suae, et prínċipem òmnis possessiónis suae, 
\verseNum{22}~ut erudíret prínċipes ejus sicut semet ìpsum, et sènes ejus prudènçiam doċéret. 
\verseNum{23}~Et intrávit Ìsrahel in Aeġỳptum, et Jacóbus àccola fuit in terra Ċami. 
\verseNum{24}~Et auxit pòpulum suum vehemènter, et firmávit eum super inimícos ejus. 
\verseNum{25}~Convèrtuit cor eórum ut odírent pòpulum ejus, et dolum fàċerent in sèrvos ejus. 
\verseNum{26}~misit Moýsem servum suum~; Aharónem, quem elégit ìpsum. 
\verseNum{27}~Pósuit in eïs vèrba signórum suórum, et prodiġiórum in terra Ċami. 
\verseNum{28}~misit tènebras, et obscurávit, et non exaċerbávit sermónes suos. 
\verseNum{29}~Convertuit aquas eórum in sángüinem, et occáesit pìsċes eórum. 
\verseNum{30}~Édidit terra eórum ranas in penetrálibus reġum ipsórum. 
\verseNum{31}~Dixit, et vënit chynómia et ċìnifes in òmnibus fínibus eórum. 
\verseNum{32}~Pósuit plúvias eórum in grándinem~; ìgnem comburèntem in terra ipsórum. 
\verseNum{33}~Et percùssit víneas eórum et ficùlneas eórum, et contrívit lìgnum fínium eórum. 
\verseNum{34}~Dixit, et venit locústa et brúchus, cujus non èrat númerus~; 
\verseNum{35}~et comédit omne foenum in terra eórum, et comédit òmnem fructum terrae eórum. 
\verseNum{36}~Et percùssit omne primoġènitum in terra eórum~; primìçias òmnis labóris eórum. 
\verseNum{37}~Et edúxit eos cum arġènto et auro, et non èrat in tribúbus eórum infìrmus. 
\verseNum{38}~Laetáta est Aeġỳptus in profecçióne eórum, quia incúbuit tìmor eórum super eos. 
\verseNum{39}~Expándit nubem in protecçiónem eórum, et ìgnem ut luċéret eïs per nòctem. 
\verseNum{40}~Petivérunt, et vënit cotùrnix, et pane ċaeli saturávit eos. 
\verseNum{41}~Dirúpit pètram, et fluxérunt aquae~; abivérunt in sicco flúmina~; 
\verseNum{42}~Quóniam mèmor fuit vèrbi sancti sui, quod hábuit ad Abrahámum púerum suum. 
\verseNum{43}~Et edúxit pòpulum suum in exsultaçióne, et eléctos suos in laetìçia. 
\verseNum{44}~Et dèdit illïs reġiónes ġèntium, et labóres populórum possedérunt~; 
\verseNum{45}~ut custódiant justificaçiónes ejus, et leġem ejus requírant. 
}
