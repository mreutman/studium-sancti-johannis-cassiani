\psalmChapterWithInscription{}
{\verseNumInscription{1} }
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{M}emènto, Dòmine, Davídis et òmnis mansuetúdinis ejus, 
\verseNum{2}~sicut jurávit Dòmino~; votum vovit Deo Jacóbi, 
\verseNum{3}~«~Si introívero in tabernáculum domüs meae, si asċèndero in lèctum strati mei, 
\verseNum{4}~si dèdero sòmnum òculïs meïs, et pálpebrïs meïs dormitaçiónem, 
\verseNum{5}~et rèquiem tempóribus meïs, donec invèniam locum Dòmino, tabernáculum Deo Jacóbi.~»
\verseNum{6}~Ecċe audívimus eam in Ephráta~; invénimus eam in campïs sỳlvae. 
\verseNum{7}~Introíbimus in tabernáculum ejus~; adorábimus in loco, ubi stetérunt pèdes ejus. 
\verseNum{8}~Sùrge Dòmine in rèquiem tuam, tu et arca sanctificaçiónis tuae. 
\verseNum{9}~Saċerdótes tui induàntur justìçiam, et sancti tui exsùltent. 
\verseNum{10}~Pròpter Davídem servum tuum, non avèrtas fàċiem christi tui. 
\verseNum{11}~Jurávit Dòminus Davídi veritátem, et non frustrábitur eam, «~De fructu vèntris tui ponem super sedem tuam. 
\verseNum{12}~Si custodíverint fílii tui testamèntum meum, et testimónia mea haec quae doċébo eos, et fílii eórum usque in sáeculum sědébunt super sedem tuam.~»
\verseNum{13}~Quóniam elégit Dòminus Siónem~; elégit eam in habitaçiónem sìbi. 
\verseNum{14}~Haec rèquies mea in sáeculum sáeculi~; hic habitábo, quóniam elégi eam. 
\verseNum{15}~Vìduam ejus benedícens benedícem~; páuperes ejus saturábo pánibus. 
\verseNum{16}~Saċerdótes ejus ìnduem salutári, et sancti ejus exsultaçióne exsultábunt. 
\verseNum{17}~Illüc prodúcem cornu Davídi~; parávi luċèrnam christo meo. 
\verseNum{18}~Inimícos ejus ìnduem confuzióne~; super ìpsum autem efflorébit sanctificáçio mea. 
}
