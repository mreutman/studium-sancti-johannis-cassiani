\psalmChapterInscription{65}
{\verseNumInscription{1}~In finem cànticum psalmi resurrecçiónis}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{J}ubiláte Deo, òmnis terra~; 
\verseNum{2}~psalmum dícite nómini ejus~; date glóriam laudi ejus. 
\verseNum{3}~Dícite Deo, «~Quam terribìlia sunt ópera tua, Dòmine. In multitúdine virtútis tuae mentièntur tìbi inimíci tui. 
\verseNum{4}~Òmnis terra adóret te, et psallat tìbi~; psalmum dícat nómini tuo.~»
\verseNum{5}~Veníte, et vidéte ópera Dei~; terrìbilis in consìliïs super fílios hòminum. 
\verseNum{6}~Qui convèrtit mare in áridam~; in flúmine pertransíbunt pède~; ìbi laetábimur in ìpso. 
\verseNum{7}~Qui dominátur in virtúte sua in aetèrnum~; òculi ejus super ġèntes respìċiunt~; qui exásperant non exaltèntur in semet ìpsïs. 
\verseNum{8}~Benedícite, ġèntes, Deum nostrum, et audítam fàċite voċem laudis ejus, 
\verseNum{9}~qui pósuit ànimam meam ad vitam, et non dèdit in commoçiónem pèdes meos. 
\verseNum{10}~Quóniam probavìsti nos, Deus~; ìgne nos examinavìsti, sicut examinátur arġèntum. 
\verseNum{11}~Induxìsti nos in làqueum~; posuìsti tribulaçiónes in dorso nostro~; 
\verseNum{12}~imposuìsti hòmines super cápita nostra. Transívimus per ìgnem et aquam, et eduxìsti nos in refrigèrium. 
\verseNum{13}~Introíbo in domum tuam in holocaustïs~; reddem tìbi vota mea, 
\verseNum{14}~quae distinxérunt lábia mea, et locútum est os meum in tribulaçióne mea. 
\verseNum{15}~Holocausta medulláta òfferem tìbi cum inċènso aríetum~; òfferem tìbi boves cum hircïs. 
\verseNum{16}~Veníte, audíte, et narrábo, òmnes qui timétis Deum, quanta feċit ànimae meae. 
\verseNum{17}~Ad ìpsum ore meo clamávi, et exaltávi sub lìngüa mea. 
\verseNum{18}~iniquitátem si aspèxi in corde meo, non exáudiet Dòminus. 
\verseNum{19}~Proptèrea exaudívit Deus, et attèndit voċi deprecaçiónis meae. 
\verseNum{20}~Benedìctus Deus, qui non amóvit oraçiónem meam et mizericórdiam suam a me. 
}
