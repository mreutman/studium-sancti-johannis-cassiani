\psalmChapterInscription{120}
{\verseNumInscription{1}~Cànticum gràduum}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\firstLetter{L}evávi òculos meos in mòntes, ùnde vèniet auxìlium mìhi.
\verseNum{2}~Auxìlium meum a Dòmino, qui feċit ċaelum et tèrram.
\verseNum{3}~Non det in commoçiónem pèdem tuum, néque dormítet qui custódit te.
\verseNum{4}~Ècċe non dormitábit néque dórmiet qui custódit Israélem.
\verseNum{5}~Dòminus custódit te, Dòminus protécçio tua super mànum dèxteram tuam.
\verseNum{6}~Per diem sol non uret te, néque luna per nòctem.
\verseNum{7}~Dòminus custódit te ab òmni màlo~; custódiat ànimam tuam Dòminus.
\verseNum{8}~Dòminus custódiat intróitum tuum, et èxitum tuum, ex höc nunc, et ùsque in sáeculum.}
