\psalmChapterWithInscription{}
{\verseNumInscription{1} }
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{1}~Laudáte Dòminum de ċaelïs~; laudáte eum in exċèlsïs. 
\verseNum{2}~Laudáte eum, òmnes ànġeli ejus~; laudáte eum, òmnes virtútes ejus. 
\verseNum{3}~Laudáte eum, sol et luna~; laudáte eum, òmnes stellae et lumen. 
\verseNum{4}~Laudáte eum, ċaeli ċaelórum, et aqua quae super ċaelos sunt. 
\verseNum{5}~Laudent nomen Dòmini~; quia ìpse dixit, et fàcta sunt~; ìpse mandávit, et creáta sunt. 
\verseNum{6}~Statúvit ea in aetèrnum, et in sáeculum sáeculi~; praeċèptum pósuit, et non praeteríbit. 
\verseNum{7}~Laudáte Dòminum de terra, dracónes et òmnes abỳssi, 
\verseNum{8}~ìgnis, gràndo, nìx, glàċia, spíritus proċèllae, quae fàċiunt vèrbum ejus, 
\verseNum{9}~mòntes et òmnes colles, lìgna fructìfera et òmnes cédri, 
\verseNum{10}~béstiae et univèrsa pècora, serpèntes et vulácres pennátae, 
\verseNum{11}~reġes terrae et òmnes pòpuli, prínċipes et òmnes júdices terrae, 
\verseNum{12}~júvenes et vìrġines, sènes cum junióribus, 
\verseNum{13}~laudent nomen Dòmini, quia exaltátum est nomen ejus sólius~; confèssio ejus super ċaelum et terram. 
\verseNum{14}~Et exaltávit cornu pòpuli sui~; hỳmnus òmnibus sanctïs ejus, fíliïs Izrahélis, pòpulo appropinquànti sìbi. 
}
