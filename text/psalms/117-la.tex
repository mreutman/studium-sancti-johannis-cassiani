\psalmChapterInscription{117}
{\verseNumInscription{1}~Allelúja}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\firstLetter{C}onfitémini Dòmino, quóniam bonus, quóniam in sáeculum mizericórdia ejus.
\verseNum{2}~Dicat nunc Ìsraël quóniam bonus, quóniam in sáeculum mizericórdia ejus.
\verseNum{3}~Dicat nunc domus Aharónis, quóniam in sáeculum mizericórdia ejus.
\verseNum{4}~Dicant nunc qui tìment Dòminum, quóniam in sáeculum mizericórdia ejus.
\verseNum{5}~De tribulaçióne invocávi Dòminum, et exaudívit me in latitúdinem Dòminus.
\verseNum{6}~Dòminus mìhi adjútor~; non timébo quid fàċiat mìhi hòmo.
\verseNum{7}~Dòminus mìhi adjútor, et ego despìċiem inimícos meos.
\verseNum{8}~Bonum est confídere in Dòmino, quam confídere in hòmine.
\verseNum{9}~Bonum est speráre in Dòmino, quam speráre in prinċìpibus.
\verseNum{10}~Òmnes ġèntes çircumivérunt me, et in nómine Dòmini quia ùltus sum in eos.
\verseNum{11}~Çircumdàntes çircumdedérunt me, et in nómine Dòmini quia ùltus sum in eos.
\verseNum{12}~Çircumdedérunt me sicut apes, et exarsérunt sicut ìgnis in spinïs, et in nómine Dòmini quia ùltus sum in eos.
\verseNum{13}~Impùlsus evèrsus sum ut càderem, et Dòminus susċépit me.
\verseNum{14}~Fortitúdo mea et laus mea Dòminus, et fàctus est mìhi in salútem.
\verseNum{15}~Vox exsultaçiónis et salútis in tabernáculïs justórum. dèxtera Dòmini feċit virtútem.
\verseNum{16}~dèxtera Dòmini exaltávit me~; dèxtera Dòmini feċit virtútem.
\verseNum{17}~Non mórier, sed vivem, et narrábo ópera Dòmini.
\verseNum{18}~Castígans castigávit me Dòminus, et morti non trádidit me.
\verseNum{19}~Aperíte mìhi portas justìçiae~; ingrèssus in eas confitébor Dòmino.
\verseNum{20}~Haec porta Dòmini~; justi intrábunt in eam.
\verseNum{21}~Confitébor tìbi quóniam exaudivìsti me, et fàctus es mìhi in salútem.
\verseNum{22}~Lápidem quem reprobavérunt aedificàntes, hic fàctus est in càput ànguli.
\verseNum{23}~A Dòmino fàctum est ìstud, et est mirábile in òculïs nostrïs.
\verseNum{24}~Haec est dies quam feċit Dòminus~; exsultémus, et laetémur in ea.
\verseNum{25}~O Dòmine, sàlvum me fac. O Dòmine, bène prosperáre.
\verseNum{26}~Benedìctus qui vènit in nómine Dòmini. Benedíximus vobïs de domu Dòmini.
\verseNum{27}~Deus Dòminus, et illúxit nobïs. Constitúite diem solèmnem in condènsïs, ùsque ad cornu altáris.
\verseNum{28}~Deus meus es tu, et confitébor tìbi~; Deus meus es tu, et exaltábo te. Confitébor tìbi quóniam exaudivìsti me, et fàctus es mìhi in salútem.
\verseNum{29}~Confitémini Dòmino, quóniam bonus, quóniam in sáeculum mizericórdia ejus.}
