\psalmChapterInscription{87}
{\verseNumInscription{1}~Cànticum psalmi fíliīs Córachi in finem pro Mahalátho ad respondèndum intelléctus Hemáni Esrahítae}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseNum{2}~\firstLetter{D}òmine, Deus salútis meae, in die clamávi, et nòcte coram te.
\verseNum{3}~Intret in conspèctu tuo oráçio mea~; inclína aurem tuam ad prèċem meam,
\verseNum{4}~quia repléta est malīs ànima mea, et vita mea infèrno appropinquávit.
\verseNum{5}~Aestimátus sum cum desċendèntibus in lacum~; fàctus sum sicut hòmo sine adjutório,
\verseNum{6}~líberus inter mórtuos, sicut vulneráti dormièntes in sepùlchrīs, quórum non es mèmor àmplius, et ìpsi de mànu tua repulsi sunt.
\verseNum{7}~Posuérunt me in lacu inferióri, in tenebrósīs, et in ùmbra mortis.
\verseNum{8}~Super me confirmátus est furor tuus, et òmnes fluctūs tuos induxìsti super me.
\verseNum{9}~Lònge feċìsti notos meos a me~; posuérunt me abominaçiónem sìbi. Tráditus sum, et non egrediébar~;
\verseNum{10}~òculi mei langüérunt prae inòpia. Clamávi ad te Dòmine tota die~; expàndi ad te mànūs meas.
\verseNum{11}~¿~Numquid mórtuīs fàċies mirabìlia, aut mèdici susċitábunt et confitebùntur tìbi~?
\verseNum{12}~¿~Numquid narrábit àliquis in sepùlchro mizericórdiam tuam, et veritátem tuam in perdiçióne~?
\verseNum{13}~¿~Numquid cognoscèntur in tènebrīs mirabìlia tua, et justìçia tua in tèrra obliviónis~?
\verseNum{14}~Et ego ad te, Dòmine, clamávi, et mane oráçio mea praevèniet te.
\verseNum{15}~¿~Ut quid, Dòmine, repèllis oraçiónem meam~; avèrtis fàċiem tuam a me~?
\verseNum{16}~Pauper sum ego, et in labóribus ab juventúte mea~; exaltátus autem, humiliátus sum et conturbátus.
\verseNum{17}~In me transivérunt irae tuae, et terróres tui conturbavérunt me.
\verseNum{18}~Çircumdedérunt me sicut aqua tota die~; çircumdedérunt me sìmul.
\verseNum{19}~Elongavìsti a me amícum et pròximum, et notos meos a mizèria.}
