\psalmChapterInscription{102}
{\verseNumInscription{1}~Ìpsi Davídis}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\verseFirstLetter{B}enedíce, ànima mea, Dòmino, et òmnia quae intra me sunt nómini sancto ejus.
\verseNum{2}~Benedíce, ànima mea, Dòmino, et noli oblivísci òmnes retribuçiónes ejus.
\verseNum{3}~Qui propiçiátur òmnibus iniquitátibus tuïs~; qui sanat òmnes infirmitátes tuas.
\verseNum{4}~Qui rèdimit de intèritu vitam tuam~; qui corónat te in mizericórdia et mizeraçiónibus.
\verseNum{5}~Qui rèplet in bonïs desidèrium tuum~; renovábitur ut àquilae juvèntüs tua.
\verseNum{6}~Fáċiens mizericórdias Dòminus, et judìċium òmnibus injúriam patièntibus.
\verseNum{7}~Notas feċit vias suas Moýsi~; fíliïs Israhélis voluntátes suas.
\verseNum{8}~Mizerátor et mizèricors Dòminus~; longánimis et mùltum mizèricors.
\verseNum{9}~Non in perpétuum irascétur, néque in aetèrnum comminábitur.
\verseNum{10}~Non secùndum peccáta nostra feċit nobïs, néque secùndum iniquitátes nostras retrìbuit nobïs.
\verseNum{11}~Quóniam secùndum altitúdinem ċaeli a terra, corroborávit mizericórdiam suam super timèntes se.
\verseNum{12}~Quantum dìstat ortus ab ocċidènte, longe feċit a nobïs iniquitátes nostras.
\verseNum{13}~Quómodo mizerétur pàter filiórum, mizèrtus est Dòminus timèntibus se.
\verseNum{14}~Quóniam ìpse cognóvit figmèntum nostrum, recordátus est quóniam pùlvis sùmus.
\verseNum{15}~hòmo, sicut foenum dies ejus, tàmquam flos agri sic efflorébit.
\verseNum{16}~Quóniam spíritus pertransívit in illo, et non subsìstet, et non cognóscet àmplius locum suum.
\verseNum{17}~Mizericórdia autem Dòmini ab aetèrno et usque in aetèrnum super timèntes eum, et justìçia ìllius in fílios filiórum,
\verseNum{18}~hïs qui servant testamèntum ejus, et mèmores sunt mandatórum ìpsius ad faċièndum ea.
\verseNum{19}~Dòminus in ċaelo parávit sedem suam, et regnum ìpsius òmnibus dominábitur.
\verseNum{20}~Benedícite Dòmino, òmnes ànġeli ejus~; potèntes virtúte, faċièntes vèrbum ìllius, ad audièndam voċem sermónum ejus.
\verseNum{21}~Benedícite Dòmino, òmnes virtútes ejus, minìstri ejus, qui fàċitis voluntátem ejus.
\verseNum{22}~Benedícite Dòmino, òmnia ópera ejus~; in òmni loco dominaçiónis ejus, benedíċe, ànima mea, Dòmino.
}
