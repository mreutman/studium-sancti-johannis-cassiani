\psalmChapterInscriptionBeginDivision{118}
{\verseNumInscription{1}~Allelúja}
{%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\firstLetter{B}eáti immaculáti in via, qui àmbulant in leġe Dòmini.
\verseNum{2}~Beáti qui scrutàntur testimónia ejus~; in toto corde exquírent eum.
\verseNum{3}~Non ènim qui operàntur iniquitátem in viīs ejus ambulavérunt.
\verseNum{4}~Tu mandavìsti mandáta tua custodíri nìmis.
\verseNum{5}~Útinam dirigàntur viae meae ad custodièndas justificaçiónes tuas.
\verseNum{6}~Tunc non confùndar, quum perspèxero in òmnibus mandátīs tuīs.
\verseNum{7}~Confitébor tìbi in direcçióne cordis, in eo quod dìdici judìċia justìçiae tuae.
\verseNum{8}~Justificaçiónes tuas custódiem~; non me derelìnquas usquequáque.
\verseNum{9}~¿~In quo còrrigit adolesċènçior viam suam~? In custodièndo sermónes tuos.
\verseNum{10}~In toto corde meo exquisívi te~; ne repèllas me a mandátīs tuīs.
\verseNum{11}~In corde meo abscòndui elóquia tua, ut non peccem tìbi.
\verseNum{12}~Benedìctus es, Dòmine~; dòċe me justificaçiónes tuas.
\verseNum{13}~In lábiīs meīs pronunçiávi òmnia judìċia oris tui.
\verseNum{14}~In via testimoniórum tuórum delectátus sum, sicut in òmnibus divìtiīs.
\verseNum{15}~In mandátīs tuīs exerċébor, et considerábo vias tuas.
\verseNum{16}~In justificaçiónibus tuīs meditábor~; non oblivíscer sermónes tuos.
\verseNum{17}~Retrìbue servo tuo~; vivìfica me, et custódiem sermónes tuos.
\verseNum{18}~Revéla òculos meos, et considerábo mirabìlia de leġe tua.
\verseNum{19}~Ìncola ego sum in tèrra~; non abscòndas a me mandáta tua.
\verseNum{20}~Concupívit ànima mea desideráre justificaçiónes tuas in òmni tèmpore.
\verseNum{21}~Increpavìsti supèrbos~; maledìcti qui declínant a mandátīs tuīs.
\verseNum{22}~Aufer a me oppróbrium et contèmptum, quia testimónia tua exquisívi.
\verseNum{23}~Ètenim sedérunt prínċipes, et advèrsum me loquebàntur~; sèrvus autem tuus exerċebátur in justificaçiónibus tuīs.
\verseNum{24}~Nam et testimónia tua meditáçio mea est, et consìlium meum justificaçiónes tuae.
\verseNum{25}~Adhaesit pavimènto ànima mea~; vivìfica me secùndum vèrbum tuum.
\verseNum{26}~Vias meas enunçiávi, et exaudivìsti me~; dòċe me justificaçiónes tuas.
\verseNum{27}~Viam justificaçiónum tuárum ìnstrue me, et exerċébor in mirabìlibus tuīs.
\verseNum{28}~Dormitávit ànima mea prae táedio~; confìrma me in vèrbīs tuīs.
\verseNum{29}~Viam iniquitátis ámove a me, et de leġe tua mizerére mei.
\verseNum{30}~Viam veritátis elégi~; judìċia tua non sum oblítus.
\verseNum{31}~Adháesi testimóniīs tuīs, Dòmine~; noli me confùndere.
\verseNum{32}~Viam mandatórum tuórum cucùrri, quum dilatavìsti cor meum.
\verseNum{33}~Leġem pone mìhi, Dòmine, viam justificaçiónum tuárum, et exquíram eam sèmper.
\verseNum{34}~Da mìhi intelléctum, et scrutábor leġem tuam, et custódiem illam in toto corde meo.
\verseNum{35}~Dedúce me in sémitam mandatórum tuórum, quia ìpsam vólui.
\verseNum{36}~Inclína cor meum in testimónia tua, et non in avarìçiam.
\verseNum{37}~Avèrte òculos meos ne vìdeant vanitátem~; in via tua vivìfica me.
\verseNum{38}~Státue servo tuo elóquium tuum in timóre tuo.
\verseNum{39}~Àmputa oppróbrium meum quod suspicátus sum, quia judìċia tua jucùnda.
\verseNum{40}~Ècċe concupívi mandáta tua~; in aequitáte tua vivìfica me.
\verseNum{41}~Et vèniat super me mizericórdia tua, Dòmine~; salutáre tuum secùndum elóquium tuum.
\verseNum{42}~Et respondébo exprobràntibus mìhi vèrbum, quia sperávi in sermónibus tuīs.
\verseNum{43}~Et ne áuferas de ore meo vèrbum veritátis usquequáque, quia in judìċiīs tuīs supersperávi.
\verseNum{44}~Et custódiem leġem tuam sèmper, in sáeculum et in sáeculum sáeculi.
\verseNum{45}~Et ambulábam in latitúdine, quia mandáta tua exquisívi.
\verseNum{46}~Et loquébar in testimóniīs tuīs in conspèctu reġum, et non confundébar.
\verseNum{47}~Et meditábar in mandátīs tuīs, quae diléxi.
\verseNum{48}~Et levávi mànūs meas ad mandáta tua, quae diléxi, et exerċébar in justificaçiónibus tuīs.
\verseNum{49}~Mèmor èsto vèrbi tui servo tuo, in quo mìhi spem dedìsti.
\verseNum{50}~Haec me consoláta est in humilitáte mea, quia elóquium tuum vivificávit me.
\verseNum{51}~Supèrbi iníque agébant usquequáque, a leġe autem tua non declinávi.
\verseNum{52}~Mèmor fui judiċiórum tuórum a sáeculo Dòmine, et consolátus sum.
\verseNum{53}~Defècçio tènuit me, pro peccatóribus derelinquèntibus leġem tuam.
\verseNum{54}~Cantábiles mìhi èrant justificaçiónes tuae in loco peregrinaçiónis meae.
\verseNum{55}~Mèmor fui nòcte nóminis tui Dòmine, et custodívi leġem tuam.
\verseNum{56}~Haec fàcta est mìhi, quia justificaçiónes tuas exquisívi.
\verseNum{57}~Pórçio mea, Dòmine~; dixi custodíre leġem tuam.
\verseNum{58}~Deprecátus sum fàċiem tuam in toto corde meo~; mizerére mei secùndum elóquium tuum.
\verseNum{59}~Coġitávi vias meas, et convèrtui pèdes meos in testimónia tua.
\verseNum{60}~Parátus sum, et non sum turbátus, ut custódiam mandáta tua.
\verseNum{61}~Funes peccatórum çircumplèxi sunt me, et leġem tuam non sum oblítus.
\verseNum{62}~Mèdia nòcte surgébam, ad confitèndum tìbi super judìċia justificaçiónis tuae.
\verseNum{63}~Pártiċeps ego sum òmnium timèntium te, et custodièntium mandáta tua.
\verseNum{64}~Mizericórdia tua, Dòmine, plena est tèrra~; justificaçiónes tuas dòċe me.
\verseNum{65}~Bonitátem feċìsti cum servo tuo, Dòmine, secùndum vèrbum tuum.
\verseNum{66}~Bonitátem et disċiplínam et sciènçiam dòċe me, quia mandátīs tuīs crédidi.
\verseNum{67}~Priùsquam humiliárer ego delíqui~; proptèrea elóquium tuum custodívi.
\verseNum{68}~Bonus es tu, et in bonitáte tua dòċe me justificaçiónes tuas.
\verseNum{69}~Multiplicáta est super me iníquitas superbórum~; ego autem in toto corde meo scrutábor mandáta tua.
\verseNum{70}~Coagulátum est sicut lac cor eórum~; ego vero leġem tuam meditátus sum.
\verseNum{71}~Bonum mìhi quia humiliavìsti me, ut dìscam justificaçiónes tuas.
\verseNum{72}~Bonum mìhi lex oris tui, super mìllia auri et arġènti.}
